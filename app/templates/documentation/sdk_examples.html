SDK Code Examples

Basic Initialization
// Initialize the SDK
const agent = new AgentSDK({
    apiKey: 'your-api-key-id:your-api-key-secret',
    project: 'my-web-app',
    environment: 'production'
});

// Track a page view
agent.track('page_view', {
    page: window.location.pathname,
    title: document.title
});

// Track a button click
document.getElementById('signup-btn').addEventListener('click', () => {
    agent.track('button_click', {
        button_id: 'signup',
        button_text: 'Sign Up'
    });
});
Error Tracking
// Track caught errors
try {
    // Risky operation
    JSON.parse(invalidData);
} catch (error) {
    agent.error(error, {
        component: 'DataParser',
        action: 'parse'
    });
}

// Global error handler
window.onerror = function(msg, url, line, col, error) {
    agent.error(error || new Error(msg), {
        url: url,
        line: line,
        column: col
    });
};

Batch Processing
// Manual batching
agent.batch([
    {
        type: 'page_view',
        data: { page: '/home' }
    },
    {
        type: 'api_call',
        data: { 
            endpoint: '/users', 
            duration: 234 
        }
    }
]);

// Auto-batching (default)
// Configure batch size and interval
const agent = new AgentSDK({
    apiKey: 'your-key',
    project: 'my-app',
    batchSize: 20,
    flushInterval: 10000
});
